This code is for custom excerpt, it will display 1 image and a portion of your post with a read more button.

(Some themes have a well-designed Read More button. You can remove the Read More button in this code and enable the button from your theme instead.)


Copy and Paste this code to your functions.php


function custom_homepage_excerpt_with_image($excerpt) {
	if (is_front_page() || is_category() || is_archive()) {
		$first_image = '';
		preg_match('/<img.*?src=["\'](.*?)["\'].*?>/i', get_the_content(), $matches);
		if (isset($matches[1])) {
			$first_image = '<img src="' . esc_url($matches[1]) . '" class="post-image" style="width: 100%; max-height: auto; margin-bottom: 15px;" />';
		}

		// Get the full post content and truncate it to 40%
		$content = get_the_content();
		$word_count = str_word_count(strip_tags($content));
		$word_limit = ceil($word_count * 0.12);
		$excerpt = wp_trim_words($content, $word_limit);

		// Add "Read More" link styled as a button
		$read_more = '<a href="' . get_permalink() . '" class="read-more-button">Read More</a>';

		// Return the image, the truncated excerpt, and the "Read More" button
		return $first_image . '<p style="line-height: 1.6;">' . $excerpt . '</p>' . $read_more;
	}
	return $excerpt;
}

// Apply the filter to the excerpt
add_filter('the_excerpt', 'custom_homepage_excerpt_with_image');

This code is for meta robots, It will add noindex nofollow to your category, author and archive pages.

Copy and Paste this code to your functions.php

If you have Yoast SEO Plugin, Use this code:


function custom_noindex_robots_for_category_and_author($robots) {
    if (is_category() || is_author() || is_archive()) {
        // Set noindex, nofollow for category, archive and author pages
        $robots = 'noindex, nofollow';
    }
    return $robots;
}
add_filter('wpseo_robots', 'custom_noindex_robots_for_category_and_author');

If you have All in One SEO Plugin, Use this code:


function aioseo_filter_robots_meta($attributes)
{
    if ((is_singular('post') || is_page() || is_home() || is_front_page())) {
        $attributes['index'] = 'index';
        $attributes['follow'] = 'follow';
    } elseif (is_category() || is_author() || is_archive()) {
        $attributes['noindex'] = 'noindex';
        $attributes['nofollow'] = 'nofollow';
    }
    return $attributes;
}
add_filter('aioseo_robots_meta', 'aioseo_filter_robots_meta');

RewriteCond %{HTTPS} off
RewriteRule ^ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]

Redirect all to HTTPS


RewriteEngine On
RewriteBase /

# Redirect www to non-www (works for both HTTP and HTTPS)
RewriteCond %{HTTP_HOST} ^www\.(.*)$ [NC]
RewriteRule ^(.*)$ https://%1/$1 [R=301,L]

# Redirect HTTP to HTTPS for non-www
RewriteCond %{HTTPS} off
RewriteCond %{HTTP_HOST} !^www\. [NC]
RewriteRule ^(.*)$ https://%{HTTP_HOST}/$1 [R=301,L]

Redirect all to HTTPS WWW


RewriteEngine On
RewriteCond %{HTTPS} off [OR]
RewriteCond %{HTTP_HOST} !^www\. [NC]
RewriteCond %{HTTP_HOST} ^(?:www\.)?(.+)$ [NC]
RewriteRule ^ https://www.%1%{REQUEST_URI} [L,NE,R=301]

Redirect all to HTTP


RewriteEngine On
RewriteBase /

# Redirect www to non-www (any domain)
RewriteCond %{HTTP_HOST} ^www\.(.+)$ [NC]
RewriteRule ^(.*)$ http://%1/$1 [R=301,L]

# Redirect HTTPS to HTTP (only for non-www)
RewriteCond %{HTTPS} on
RewriteCond %{HTTP_HOST} !^www\. [NC]
RewriteRule ^(.*)$ http://%{HTTP_HOST}/$1 [R=301,L]

This code is for the featured image in block themes or you can also try on other normal themes.

If you don’t want to upload a featured image but your theme requires one for the layout,
This code checks if a post has a featured image. If not, it extracts the first image from the post content and uses it as the thumbnail.

Add this code to your theme's functions.php file.


function get_first_image_from_post($post_id) {
    $post = get_post($post_id);
    if (!$post) return '';

    // Extract images from post content
    preg_match('/<img.*?src=["\'](.*?)["\']/', $post->post_content, $matches);
    
    return !empty($matches[1]) ? $matches[1] : ''; // Return first image URL if found
}

function custom_post_thumbnail($html, $post_id, $post_thumbnail_id, $size, $attr) {
    if (has_post_thumbnail($post_id)) {
        return $html; // If featured image exists, return it
    }

    $first_image = get_first_image_from_post($post_id);
    if ($first_image) {
        return '<img src="' . esc_url($first_image) . '" class="wp-post-image" alt="">';
    }

    return $html; // Return default output if no image is found
}

add_filter('post_thumbnail_html', 'custom_post_thumbnail', 10, 5);
